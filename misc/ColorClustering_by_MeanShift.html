def show_colors(colors):
    cs = len(colors)
    H,W = 100, 40
    m = np.zeros( (H, W*cs,3) )
    print(m.shape)
    for i, c in enumerate(colors):
        m[:,i*W:(i+1)*W] = c
    io.imsave("colors.png", m.astype(np.uint8) )
    return m.astype(np.uint8)
